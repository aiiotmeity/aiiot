const calculateDistance = (lat1, lon1, lat2, lon2) => {
    const R = 6371; // km
    const dLat = (lat2 - lat1) * Math.PI / 180;
    const dLon = (lon2 - lon1) * Math.PI / 180;
    const a = Math.sin(dLat / 2) * Math.sin(dLat / 2) + Math.cos(lat1 * Math.PI / 180) * Math.cos(lat2 * Math.PI / 180) * Math.sin(dLon / 2) * Math.sin(dLon / 2);
    const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
    return R * c;
};
const createAqiIcon = (aqi) => {
    const color = getAQIColor(aqi);
    const iconHtml = `<div style="background-color: ${color};" class="aqi-marker-icon"><div class="aqi-marker-icon-inner">${Math.round(aqi) || 'N/A'}</div></div>`;
    return window.L.divIcon({ html: iconHtml, className: 'custom-div-icon', iconSize: [32, 32], iconAnchor: [16, 32] });
};

const MapPage = () => {
    const [mapInstance, setMapInstance] = useState(null);
    const [stations, setStations] = useState({});
    const [selectedStationId, setSelectedStationId] = useState(null);
    const [isLoading, setIsLoading] = useState(true);
    const [error, setError] = useState(null);
    const [isPanelOpen, setIsPanelOpen] = useState(true);
    
    // State for User Location
    const [userLocation, setUserLocation] = useState(null);
    const [nearestStation, setNearestStation] = useState(null);
    const [isLocating, setIsLocating] = useState(false);

    const mapRef = useRef(null);
    const markersRef = useRef({});
    const userLocationMarkerRef = useRef(null);
    const { user } = useAuth();
    const API_BASE_URL = process.env.REACT_APP_API_URL || 'http://localhost:8000';

    useEffect(() => {
        let map;
        if (mapRef.current && !mapRef.current._leaflet_id) {
            map = window.L.map(mapRef.current, { zoomControl: false }).setView([10.176, 76.430], 13);
            window.L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png').addTo(map);
            window.L.control.zoom({ position: 'topleft' }).addTo(map);
            setMapInstance(map);
        }
        return () => { if (map) map.remove(); };
    }, []);
    
    const trackUserLocation = useCallback(() => {
        setIsLocating(true);
        setNearestStation(null); // Reset previous results
        if (navigator.geolocation) {
            navigator.geolocation.getCurrentPosition(
                (position) => {
                    const loc = { lat: position.coords.latitude, lng: position.coords.longitude };
                    setUserLocation(loc);
                    if (mapInstance) {
                        mapInstance.setView(loc, 14);
                        if (userLocationMarkerRef.current) {
                            userLocationMarkerRef.current.setLatLng(loc);
                        } else {
                            userLocationMarkerRef.current = window.L.marker(loc).addTo(mapInstance);
                        }
                    }
                    setIsLocating(false);
                },
                (err) => {
                    alert("Could not get your location. Please enable location services.");
                    setIsLocating(false);
                }
            );
        }
    }, [mapInstance]);

    const fetchRealtimeData = useCallback(async () => {
        setIsLoading(true);
        try {
            const response = await fetch(`${API_BASE_URL}/api/map/realtime/`);
            const data = await response.json();
            setStations(data.stations);
            if (!selectedStationId && Object.keys(data.stations).length > 0) {
                setSelectedStationId(Object.keys(data.stations)[0]);
            }
        } catch (err) {
            setError(err.message);
        } finally {
            setIsLoading(false);
        }
    }, [API_BASE_URL, selectedStationId]);

    useEffect(() => {
        fetchRealtimeData();
        const interval = setInterval(fetchRealtimeData, 60000);
        return () => clearInterval(interval);
    }, [fetchRealtimeData]);

    // Effect to calculate nearest station after location is found
    useEffect(() => {
        if (userLocation && Object.keys(stations).length > 0) {
            let nearestDist = Infinity, nearestId = null;
            Object.entries(stations).forEach(([id, station]) => {
                const dist = calculateDistance(userLocation.lat, userLocation.lng, station.station_info.lat, station.station_info.lng);
                if (dist < nearestDist) { nearestDist = dist; nearestId = id; }
            });
            setNearestStation({ id: nearestId, distance: nearestDist });
        }
    }, [userLocation, stations]);
    
    useEffect(() => {
        if (!mapInstance || Object.keys(stations).length === 0) return;
        Object.values(markersRef.current).forEach(marker => marker.remove());
        markersRef.current = {};
        Object.entries(stations).forEach(([id, station]) => {
            const marker = window.L.marker([station.station_info.lat, station.station_info.lng], { icon: createAqiIcon(station.highest_sub_index) }).addTo(mapInstance);
            marker.on('click', () => {
                setSelectedStationId(id);
                setIsPanelOpen(true);
                mapInstance.setView([station.station_info.lat, station.station_info.lng], 15);
            });
            markersRef.current[id] = marker;
        });
    }, [mapInstance, stations, selectedStationId]);

    const selectedStationData = stations[selectedStationId];
    // ... (Your forecast chart logic and other constants can remain here)

    return (
        <div className="map-page">
            <nav className="navbar">{/* ... your navbar JSX ... */}</nav>
            <div className="main-content">
                <div className="map-container">
                    <div id="map" ref={mapRef} className="map-element"></div>
                    <button onClick={trackUserLocation} className="my-location-btn" title="Find My Location">
                        <i className="fas fa-crosshairs"></i>
                    </button>
                </div>
                <div className={`details-panel ${isPanelOpen ? 'open' : ''}`}>
                    <button className="panel-toggle" onClick={() => setIsPanelOpen(!isPanelOpen)}>{/* ... */}</button>
                    <div className="panel-header">{/* ... your station selector ... */}</div>
                    <div className="panel-content">
                        {/* --- NEW, CORRECTED User Location Panel Logic --- */}
                        {isLocating && (
                             <div className="user-location-panel">
                                <div className="user-location-header"><i className="fas fa-spinner fa-spin"></i><span>Locating...</span></div>
                             </div>
                        )}
                        {nearestStation && (
                            <div className="user-location-panel">
                                {nearestStation.distance < 1 ? (
                                    <>
                                        <div className="user-location-header"><i className="fas fa-street-view"></i><span>AQI at Your Location</span></div>
                                        <div className="user-location-body">
                                            <div className="user-aqi-value">{stations[nearestStation.id]?.highest_sub_index}</div>
                                            <div className="user-location-details">
                                                <span>Using data from nearest station</span><br />
                                                <span>{stations[nearestStation.id]?.station_info.name} ({nearestStation.distance.toFixed(1)} km)</span>
                                            </div>
                                        </div>
                                    </>
                                ) : (
                                    <>
                                        <div className="user-location-header"><i className="fas fa-exclamation-triangle"></i><span>Location Status</span></div>
                                        <p>You are not within 1km of the nearest sensor station, so a precise AQI for your location cannot be provided.</p>
                                        <div className="user-location-details">
                                            <span>Nearest: {stations[nearestStation.id]?.station_info.name} ({nearestStation.distance.toFixed(1)} km away)</span>
                                        </div>
                                    </>
                                )}
                            </div>
                        )}

                        {isLoading ? <div className="panel-loader"><div className="loading-spinner"></div></div> :
                        selectedStationData ? (
                            <div>
                                <h3>{selectedStationData.station_info.name}</h3>
                                {/* ... Your existing station details, pollutant metrics, and forecast chart ... */}
                            </div>
                        ) : null}
                    </div>
                </div>
            </div>
        </div>
    );
};

export default MapPage;
